5 /*
6  Multilinear interpolation.
7  Up to five dimensions.
8  These are hard coded. This is relatively easy to read, but not as general as,
9  e.g., https://github.com/parsiad/mlinterp
10  This version assumes simple arrays.
11  Pass in the number of dimensions for each array,
12  then the grid values for each dimension (arrays),
13  then the array of function values at each point,
14  then the locations of the desired points in each dimension.
15  The grid of function values is mapped to 1-D in row-major form:
16  For x,y,z indexed by i,j,k, we have f[i][j][k] ==> f[i*nz*ny + j*nz + k]
17  See LI_2D for specific description of values passed.
18  LI_3D and LI_4D are tested by comparison of the rcSLW code to the python version.
19 */

22 int get_location_dim(const int nx, const double *x, const double xP) {
23  if(xP <= x[0])
24  return 0;
25  else if(xP >= x[nx-1])
26  return nx-2;
27  else
28  return lower_bound(x, x+nx, xP) - x - 1;
29 }
